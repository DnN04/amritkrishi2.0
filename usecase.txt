CREATE DATABASE amritkrishi;

USE amritkrishi;

-- Table for farmer information and location
CREATE TABLE farmers (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    farmer_name VARCHAR(255) NOT NULL,
    latitude DECIMAL(10, 8) NOT NULL,
    longitude DECIMAL(11, 8) NOT NULL,
    location_method VARCHAR(50),
    address TEXT,
    preferred_language VARCHAR(10) DEFAULT 'en',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- See farmer with their crop recommendations
SELECT f.farmer_name, f.latitude, f.longitude, c.crop_name, c.suitability_score
FROM farmers f
LEFT JOIN crop_recommendations c ON f.id = c.farmer_id;

-- Show username - location - best crop (highest score)
USE amritkrishi;
SELECT 
    CONCAT(f.farmer_name, ' - ', 
           ROUND(f.latitude, 4), ', ', ROUND(f.longitude, 4), ' - ',
           COALESCE(c.crop_name, 'No recommendation')) AS farmer_crop_info
FROM farmers f
LEFT JOIN (
    SELECT farmer_id, crop_name, suitability_score,
           ROW_NUMBER() OVER (PARTITION BY farmer_id ORDER BY suitability_score DESC) as rn
    FROM crop_recommendations
) c ON f.id = c.farmer_id AND c.rn = 1
ORDER BY f.farmer_name;
